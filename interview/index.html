排序 #

完善以下代码，写一个平均时间复杂度低于 O(n^2) 的排序算法（空间 复杂度不作要求）：

function sort(a) {
    for (let j = 1; j < a.length; j++) {
      let key = a[j];
      let i = j - 1;
      while (i >= 0 && a[i] > key) {
	  a[i+1] = a[i];
	  i = i - 1;
      }
      a[i+1] = key;
    }
    return a;
}

缓存 #

完善下面代码，输出第 n (n=0,1,2,3,…) 个斐波那契数，要求 n 为 50 时，能够在 2秒内 返回结果

function fibonacci(n) {
    let phi = (1 + Math.sqrt(5)) / 2;
    return Math.round(Math.pow(phi, n) / Math.sqrt(5));
}

Promise #

完善 light 函数，要求运行代码之后，1 秒后打印 red，打印 red 之后的 2 秒后打印 green， 打印 green 的 3 秒后打印 yellow

function red() {
    console.log(new Date());
    console.log('red');
}
function green() {
    console.log(new Date());
    console.log('green');
}
function yellow() {
    console.log(new Date());
    console.log('yellow');
}

light(1, red).then(function(){
    return light(2, green);
}).then(function() {
    return light(3, yellow);
})

function light(timer, fn) {
    let promise = new Promise(function(resolve, reject) {
      setTimeout(() => resolve(fn), timer*1000);
    });

    return promise.then(fn);
}
